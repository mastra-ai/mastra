const agent = new Agent({
  id: "dynamic-agent",
  name: "Dynamic Agent",
  model: ({ requestContext }) => {
    const useAdvanced = requestContext.task === "complex";
    return useAdvanced
      ? "snowflake-cortex/snowflake-llama3.3-70b"
      : "snowflake-cortex/claude-fable-5";
  }
});
