const agent = new Agent({
  id: "dynamic-agent",
  name: "Dynamic Agent",
  model: ({ requestContext }) => {
    const useAdvanced = requestContext.task === "complex";
    return useAdvanced
      ? "friendli/zai-org/GLM-5"
      : "friendli/LGAI-EXAONE/EXAONE-4.0.1-32B";
  }
});
