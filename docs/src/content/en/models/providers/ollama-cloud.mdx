```typescript
});
```

```typescript
const agent = new Agent({
  id: "dynamic-agent",
  name: "Dynamic Agent",
  model: ({ requestContext }) => {
    const useAdvanced = requestContext.task === "complex";
    return useAdvanced
      ? "ollama-cloud/qwen3-vl-235b-instruct-cloud"
      : "ollama-cloud/cogito-2.1:671b-cloud";
  }
});
```
