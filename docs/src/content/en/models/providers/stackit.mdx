  name: "Dynamic Agent",
  model: ({ requestContext }) => {
    const useAdvanced = requestContext.task === "complex";
    return useAdvanced
      ? "stackit/openai/gpt-oss-120b"
      : "stackit/Qwen/Qwen3-VL-235B-A22B-Instruct-FP8";
  }
});
