const agent = new Agent({

  name: "Dynamic Agent",
  model: ({ requestContext }) => {
    const useAdvanced = requestContext.task === "complex";
    return useAdvanced
      ? "cline-pass/cline-pass/qwen3.7-plus"
      : "cline-pass/cline-pass/deepseek-v4-flash";
  }
});
