### 动态模型选择

const agent = new Agent({
  id: "dynamic-agent",
  name: "Dynamic Agent",
  model: ({ requestContext }) => {
    const useAdvanced = requestContext.task === "complex";
    return useAdvanced
      ? "minimax-coding-plan/MiniMax-M2.1"
      : "minimax-coding-plan/MiniMax-M2";
  }
});
