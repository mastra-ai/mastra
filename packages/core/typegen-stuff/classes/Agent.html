    generate

    • generate(
          messages: MessageListInput,
          args?: AgentGenerateOptions<undefined, undefined> & { output?: undefined; experimental_output?: undefined; },
      ): Promise<GenerateTextResult<any, undefined>>

      Generates text or structured responses from the agent.

      This method accepts messages and optional generation options to produce text or structured responses. It supports structured outputs via the output or experimental_output parameters, and can handle multimodal inputs (text, images, etc.).

      Note: This method will switch to use generateVNext implementation in September 2025. Use generateLegacy if you need to maintain the current behavior.

      Parameters

      • messages: MessageListInput

        The messages to send to the agent. Can be a single string, array of strings, or structured message objects with multimodal content

      • Optionalargs: AgentGenerateOptions<undefined, undefined> & { output?: undefined; experimental_output?: undefined; }

      Returns Promise<GenerateTextResult<any, undefined>>

      A promise that resolves to either text or structured output based on the schema provided

      Example

      import { z } from 'zod';
      import { ModerationProcessor, TokenLimiterProcessor } from "@mastra/core/processors";

      // Simple text generation
      const result = await agent.generate("What is the weather like today?");
      console.log(result.text);

      // With options
      const enhancedResult = await agent.generate(
        [
          { role: "user", content: "message for agent" },
          {
            role: "user",
            content: [
              { type: "text", text: "message for agent" },
              { type: "image", imageUrl: "https://example.com/image.jpg", mimeType: "image/jpeg" }
            ]
          }
        ],
        {
          temperature: 0.7,
          maxSteps: 3,
          memory: {
            thread: "user-123",
            resource: "test-app"
          },
          toolChoice: "auto",
          providerOptions: {
            openai: { reasoningEffort: "high" }
          },
          structuredOutput: {
            schema: z.object({
              sentiment: z.enum(['positive', 'negative', 'neutral']),
              confidence: z.number(),
            }),
            model: openai("gpt-4o-mini"),
            errorStrategy: 'warn',
          },
          outputProcessors: [
            new ModerationProcessor({ model: openai("gpt-4.1-nano") }),
            new TokenLimiterProcessor({ maxTokens: 1000 }),
          ],
        }
      );
